Exemplo de Código em NCL que exibe um mapa posicionado utilizando atributos left e top, e alterando sua descrição para que dure 7 segundos e desapareçã com uma transição fade. Durante esse processo ficará com uma transparência de 70%.

[code lang="xml"]

<?xml version="1.0" encoding="ISO-8859-1"?>
<!--
Author: Luiz Arão Araújo Carvalho
Site: www.redrails.com.br
Data: 07/12/2011
Contato: luizcarvalho@redrails.com.br / @LuizCarvalho

-->
<ncl id="main" xmlns="http://www.ncl.org.br/NCL3.0/EDTVProfile">
<head>
<transitionBase>
<transition id="transFade" type="fade"/>
</transitionBase>
<regionBase>
<region id="rgMainVideo" width="100%" height="100%" zIndex="0" />
<region id="rgLogo" width="128" height="38" top="10px" right="10px" zIndex="0" />
<region id="rgMapa" width="212" height="360" top="10px" left="10px" zIndex="0" />
</regionBase>

<descriptorBase>
<descriptor id="descMainVideo" region="rgMainVideo" />
<descriptor id="descLogo" region="rgLogo" />
<descriptor id="descMapa" region="rgMapa" explicitDur="7s" transIn="fade" transOut="fade" >
<descriptorParam name="transparency" value="70%"/>
</descriptor>

</descriptorBase>

</head>

<body>
<port id="pMVideo" component="mMainvideo" />
<port id="pLogo" component="mLogo" />
<port id="pMapa" component="mMapa" />

<media id="mMainvideo" src="media/video.mp4" descriptor="descMainVideo" />
<media id="mLogo" src="media/espn.png" descriptor="descLogo" />
<media id="mMapa" src="media/mapa.png" descriptor="descMapa" />
</body>
</ncl>

[/code]
